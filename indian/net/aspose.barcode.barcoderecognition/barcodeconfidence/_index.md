### मान

| नाम | कीमत | विवरण |
| --- | --- | --- |
| None | `0` | बारकोड की मान्यता विश्वास जहां कोडटेक्स्ट को सही ढंग से पहचाना नहीं गया था या बारकोड को संभावित नकली के रूप में पाया गया था |
| Moderate | `80` | कमजोर चेकसम के साथ या इसके बिना भी बारकोड (ज्यादातर 1डी बारकोड) की पहचान विश्वास। codetext में कुछ गलत पहचान हो सकती है या नकली पहचान भी हो सकती है कम है |
| Strong | `100` | रिकॉग्निशन कॉन्फिडेंस जिसकी पुष्टि रीड-सोलोमन जैसे बीसीएच कोड से की गई थी। रीड कोडटेक्स्ट या नकली पहचान में त्रुटियां नहीं होनी चाहिए |

[C#]
// मध्यम आत्मविश्वास
using (BarcodeGenerator generator = new BarcodeGenerator(EncodeTypes.Code128, "12345"))
{
    generator.Save(@"c:\test.png");
}
using (BarCodeReader reader = new BarCodeReader(@"c:\test.png", DecodeType.Code39Standard, DecodeType.Code128))
{
    foreach (BarCodeResult result in reader.ReadBarCodes())
    {
        Console.WriteLine("BarCode Type: " + result.CodeTypeName);
        Console.WriteLine("BarCode CodeText: " + result.CodeText);
        Console.WriteLine("BarCode Confidence: " + result.Confidence);
        Console.WriteLine("BarCode ReadingQuality: " + result.ReadingQuality);
    }
}
